```typescript
// line 145-167: async implementation
const _getEnvCommitFromGit = async (): Promise<EnvironmentCommitSlug | null> => {
  // get exact tag (if HEAD is tagged)
  const tag = await runGit('git describe --tags --exact-match 2>/dev/null');

  // get branch name (fallback if no tag)
  const branch = await runGit('git rev-parse --abbrev-ref HEAD 2>/dev/null');

  // pick ref (tag or branch)
  const ref = computeGitRef({ tag, branch });
  if (!ref) return null;

  // get short hash
  const hash = await runGit('git rev-parse --short HEAD 2>/dev/null');
  if (!hash) return null;

  // check if dirty
  const status = await runGit('git status --porcelain 2>/dev/null');
  const dirty = computeIsDirty({ status });

  // format slug
  return computeCommitSlug({ ref, hash, dirty });
};
```

### key behaviors

1. **tag precedence**: `computeGitRef` returns tag over branch (line 118-121)
2. **dirty detection**: `computeIsDirty` checks if status output is non-empty (line 127-128)
3. **format**: `computeCommitSlug` formats as `$ref@$hash` or `$ref@$hash+` (line 134-139)
4. **error handling**: `isExpectedGitError` allowlists known git failures (line 20-46)

## pattern.3 = rhachet bin dispatcher

**location**: `ehmpathy/rhachet/bin/`

**[REUSE]** — follow this pattern for our CLI

### structure

```
bin/
  run              # bash shim, routes to run.bun
  run.bun          # shell dispatcher for bun binaries
  run.jit          # jit fallback (tsx)
  run.bun.*.bc     # bun-compiled binaries
```

### key code (bin/run lines 1-43)

```bash
#!/usr/bin/env bash
# routes to bun (fast) or jit (flexible) dispatcher

# derive bin directory (handles symlinks from node_modules/.bin/)
BIN="${BASH_SOURCE[0]}"
while [ -L "$BIN" ]; do
  BIN_DIR="$(cd "$(dirname "$BIN")" && pwd)"
  BIN="$(readlink "$BIN")"
  [[ "$BIN" != /* ]] && BIN="$BIN_DIR/$BIN"
done
BIN_DIR="$(cd "$(dirname "$BIN")" && pwd)"

case "$1" in
  run)
    exec "$BIN_DIR/run.bun" "$@"
    ;;
  # ...
esac
```

### for sdk-environment

our CLI is simpler — single command `get commit`:
- no jit fallback needed (wisher confirmed)
- single bun binary sufficient
- bash shim routes directly to compiled binary
