# blueprint: cli binary for commit

## summary

add a CLI binary that outputs the commit slug to stdout, so consumers can call `npx sdk-environment get commit` instead of manual git commands.

- reuse extant `getEnvCommitFromGit.sync()` for core logic
- add bash shim that routes to bun-compiled binary
- semantic exit codes: 0=success, 1=malfunction, 2=constraint

## implementation notes

### bash shim (bin/run)

```bash
#!/usr/bin/env bash
# derive bin directory (handles symlinks from node_modules/.bin/)
BIN="${BASH_SOURCE[0]}"
while [ -L "$BIN" ]; do
  BIN_DIR="$(cd "$(dirname "$BIN")" && pwd)"
  BIN="$(readlink "$BIN")"
  [[ "$BIN" != /* ]] && BIN="$BIN_DIR/$BIN"
done
BIN_DIR="$(cd "$(dirname "$BIN")" && pwd)"

exec "$BIN_DIR/run.bun.sdk-environment.bc" "$@"
```

### CLI router (invoke.ts)

```typescript
import { getCommit } from './getCommit';

const [, , cmd, subcmd] = process.argv;

if (cmd === 'get' && subcmd === 'commit') {
  getCommit();
} else {
  process.stderr.write('unknown command\n');
  process.exit(2);
}
```

### get commit command (getCommit.ts)

```typescript
import { getEnvCommitFromGit } from '@/domain.operations/parsers/commit/getEnvCommitFromGit';

/**
 * .what = outputs commit slug to stdout for cli capture
 * .why = consumers need canonical commit format at deploy time
 */
export const getCommit = (): void => {
  try {
    const slug = getEnvCommitFromGit.sync();
    if (!slug) {
      process.stderr.write('error: could not determine commit (not a git repo or no ref found)\n');
      process.exit(2);
    }
    process.stdout.write(`${slug}\n`);
    process.exit(0);
  } catch (error) {
    process.stderr.write(`error: ${error instanceof Error ? error.message : String(error)}\n`);
    process.exit(1);
  }
};
```
